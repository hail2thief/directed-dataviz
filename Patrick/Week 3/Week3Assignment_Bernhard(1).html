# Select relevant metrics and compute correlation matrix
metrics <- data[, c("irst", "milex", "milper", "tpop", "upop")]
cor_matrix <- cor(metrics, use = "complete.obs")  # use complete.obs ignores missing values

# Melt the correlation matrix for ggplot2
melted_cor_matrix <- melt(cor_matrix)

# Create the heatmap
p_heatmap <- ggplot(melted_cor_matrix, aes(x = Var1, y = Var2, fill = value)) +
  geom_tile() + 
  scale_fill_gradient2(low = "blue", high = "red", mid = "white", 
                       midpoint = 0, limit = c(-1,1), space = "Lab",
                       name="Correlation") +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1)) +
  labs(title = "Correlation Heatmap of Military Metrics")

print(p_heatmap)
